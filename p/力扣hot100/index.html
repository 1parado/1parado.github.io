        // ÂèØËÉΩÂá∫Áé∞ÁöÑÊÉÖÂÜµÂõõÁßç1.ÊâæÂà∞nums[mid] = target
        // 2.Ê≤°Âá∫Áé∞ ÁõÆÊ†áÂÄºÂú®ÊúÄÂâçÈù¢

                return mid;
            }else if(nums[mid] > target){
                high = mid - 1;
                mid = (high + low)/2;

                mid = (high + low)/2;
            }
            
        }
        return high + 1;
    }
